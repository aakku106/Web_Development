JavaScript Loops Practice Set Solutions

Q1: Basic for Loop

Output:


    
Code:

    
for(let i=1;i<=10;i++){
  console.log(i);
}
    


    
Q2: while Loop (Even numbers)

    

    
let i=2;
while(i<=20){
  console.log(i);
  i+=2;
}
    


    
Q3: do...while Loop (Countdown)

    

    
let i=5;
do{
  console.log(i);
  i--;
}while(i>=1);
    


    
Q4: for...in Loop (Object keys)

    

    
let student={name:'Pramish',age:13,grade:8};
for(let key in student){
  console.log(key+': '+student[key]);
}
    


    
Q5: for...of Loop (Array values)

    

    
let fruits=['Apple','Banana','Mango','Orange'];
for(let fruit of fruits){
  console.log(fruit);
}
    


    
Q6: Break Statement

    

    
for(let i=1;i<=10;i++){
  if(i===6) break;
  console.log(i);
}
    


    
Q7: Continue Statement

    

    
for(let i=1;i<=10;i++){
  if(i===5) continue;
  console.log(i);
}
    


    
Q8: Nested Loops (3x3 Grid)

    

    
for(let i=1;i<=3;i++){
  let row='';
  for(let j=1;j<=3;j++){
    row+=i+' ';
  }
  console.log(row);
}
    


    
Q9: Combined Practice (Array of Objects)

    

    
let students=[{name:'A',age:12},{name:'B',age:14},{name:'C',age:11}];
students.forEach(s=>{
  if(s.age>12) console.log(s.name);
});
