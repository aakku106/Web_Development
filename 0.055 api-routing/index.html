📚 What is API Routing?

🎯 Simple Definition

API Routing in Next.js lets you create backend endpoints (mini servers) directly inside your Next.js app. No need for a separate Express server!

Traditional vs Next.js

❌ Traditional Setup

⚛️ React Frontend Port 3000
→
🚂 Express Server Port 5000
→
🗄️ Database MongoDB

Three separate applications!

✅ Next.js API Routes

⚡ Next.js (Frontend + Backend) Port 3000
→
🗄️ Database MongoDB

All in one application!

🚀 Your First API Route

pages/api/hello.js 
// This runs on the SERVER!
export default function handler(req, res) {
  res.status(200).json({ 
    message: 'Hello from API!',
    timestamp: new Date().toISOString()
  });
}

// Visit: http://localhost:3000/api/hello

🔑 Key Concepts

📥

req (Request)

Incoming data from user

  • req.method → GET, POST, etc.
  • req.body → Data sent
  • req.query → URL parameters
📤

res (Response)

What you send back

  • res.status(200) → Success
  • res.json({}) → Send JSON
  • res.send() → Send text

🔧 HTTP Methods

GET

Retrieve Data

Fetch data from server (like reading a book)

pages/api/users.js 
export default function handler(req, res) {
  if (req.method === 'GET') {
    const users = [
      { id: 1, name: 'Alice' },
      { id: 2, name: 'Bob' }
    ];
    
    res.status(200).json(users);
  }
}
Frontend Usage: 
const response = await fetch('/api/users');
const users = await response.json();
POST

Create Data

Send data to create something new (like writing a new page)

pages/api/users.js 
export default function handler(req, res) {
  if (req.method === 'POST') {
    const { name, email } = req.body;
    
    const newUser = {
      id: Date.now(),
      name,
      email
    };
    
    res.status(201).json(newUser);
  }
}
Frontend Usage: 
const response = await fetch('/api/users', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ name: 'Charlie', email: 'charlie@email.com' })
});
PUT

Update Data

Modify existing data (like editing a page)

pages/api/users/[id].js 
export default function handler(req, res) {
  const { id } = req.query;
  
  if (req.method === 'PUT') {
    const { name, email } = req.body;
    
    const updated = { id, name, email };
    
    res.status(200).json(updated);
  }
}
DELETE

Remove Data

Delete data (like tearing out a page)

pages/api/users/[id].js 
export default function handler(req, res) {
  const { id } = req.query;
  
  if (req.method === 'DELETE') {
    // Delete from database
    res.status(200).json({ 
      message: 'User deleted', 
      id 
    });
  }
}

📊 HTTP Status Codes

200 OK Success!
201 Created New item created
400 Bad Request Invalid data sent
404 Not Found Resource doesn't exist
500 Server Error Something broke

💡 Real-World Examples

1. User Authentication

pages/api/auth/login.js 
export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { email, password } = req.body;

  // Validate input
  if (!email || !password) {
    return res.status(400).json({ 
      error: 'Email and password required' 
    });
  }

  // Check credentials
  if (email === 'user@example.com' && password === 'password123') {
    return res.status(200).json({
      success: true,
      user: { email, name: 'John Doe' },
      token: 'jwt_token_here'
    });
  }

  res.status(401).json({ error: 'Invalid credentials' });
}

2. Database Operations

pages/api/products.js 
import { connectToDatabase } from '@/lib/mongodb';

export default async function handler(req, res) {
  const { db } = await connectToDatabase();

  if (req.method === 'GET') {
    // Fetch all products
    const products = await db
      .collection('products')
      .find({})
      .toArray();
    
    return res.status(200).json(products);
  }

  if (req.method === 'POST') {
    // Create new product
    const { name, price } = req.body;
    
    const result = await db
      .collection('products')
      .insertOne({ name, price, createdAt: new Date() });
    
    return res.status(201).json({ id: result.insertedId });
  }

  res.status(405).json({ error: 'Method not allowed' });
}

3. External API Call

pages/api/weather.js 
export default async function handler(req, res) {
  const { city } = req.query;

  if (!city) {
    return res.status(400).json({ 
      error: 'City parameter required' 
    });
  }

  try {
    const response = await fetch(
      `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=YOUR_KEY`
    );
    
    const data = await response.json();
    
    res.status(200).json({
      city: data.name,
      temperature: data.main.temp,
      description: data.weather[0].description
    });
  } catch (error) {
    res.status(500).json({ 
      error: 'Failed to fetch weather data' 
    });
  }
}

4. Dynamic Routes

pages/api/products/[id].js

  const { id } = req.query;
  
  // GET /api/products/123 → id = "123"
  
  const product = {
    id,
    name: `Product ${id}`,
    price: 99.99,
    stock: 50
  };
  
  res.status(200).json(product);
}

✅ Best Practices

✅ DO

  • ✅ Validate all input data
  • ✅ Handle errors gracefully
  • ✅ Use proper status codes
  • ✅ Secure sensitive data
  • ✅ Use environment variables
  • ✅ Implement rate limiting

❌ DON'T

  • ❌ Expose passwords/secrets
  • ❌ Skip input validation
  • ❌ Ignore error handling
  • ❌ Use synchronous operations
  • ❌ Return sensitive data
  • ❌ Forget CORS headers
