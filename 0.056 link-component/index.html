What is the Link Component?

The Link component provides instant navigation between pages without full page reload!

📚 Simple Usage

export default function Home() {
  return (
    <div>
      <h1>Welcome Home!</h1>

❌ Wrong: Using <a> Tag

// DON'T do this in Next.js
<a href="/about">About Page</a>
// This causes full page reload ⏳

✅ Correct: Using Link

// DO this in Next.js
<Link href="/about">About Page</Link>
// Fast client-side navigation ⚡

⚠️ When to Use <a> Tag

Use regular anchor tags ONLY for external links:

<a href="https://nextjs.org" target="_blank" rel="noopener noreferrer">
  Next.js Documentation
</a>

Link Component Props

href Required

The destination path or URL

<Link href="/about">About</Link>
<Link href="/blog/1">Blog Post</Link>
<Link href={`/user/${id}`}>Profile</Link>
prefetch Optional

Prefetch page in background (default: true) 

// Enable prefetch (default)
<Link href="/about" prefetch={true}>About</Link>

// Disable for heavy pages
<Link href="/dashboard" prefetch={false}>Dashboard</Link>
replace Optional

Replace history instead of push (default: false) 

// Normal navigation (adds to history)
<Link href="/page2">Next Page</Link>

// Replace current page (no back button)
<Link href="/home" replace={true}>Home</Link>
scroll Optional

Scroll to top after navigation (default: true) 

// Scroll to top (default)
<Link href="/article" scroll={true}>Article</Link>

// Keep scroll position
<Link href="/comments" scroll={false}>Comments</Link>

💡 Pro Tip: Object-Based href

<Link href={{
  pathname: '/search',
  query: { q: 'nextjs', category: 'tutorial' }
}}>
  Search Results
</Link>
// Results in: /search?q=nextjs&category=tutorial

Real-World Navigation Examples

1. Basic Navigation Bar

MyWebsite
Home About Services Blog Contact 
import Link from 'next/link';

export default function Navbar() {
  return (
    <nav>
      <Link href="/">Home</Link>
      <Link href="/about">About</Link>
      <Link href="/services">Services</Link>
      <Link href="/blog">Blog</Link>
      <Link href="/contact">Contact</Link>
    </nav>
  );
}

2. Dynamic Routes with IDs

// Blog post list
const posts = [
  { id: 1, title: 'Getting Started with Next.js' },
  { id: 2, title: 'Understanding Link Component' },
  { id: 3, title: 'Building Fast Websites' }
];

export default function BlogList() {
  return (
    <div>
      {posts.map(post => (
        <Link key={post.id} href={`/blog/${post.id}`}>
          {post.title}
        </Link>
      ))}
    </div>
  );
}

3. Active Link Pattern

'use client';
import Link from 'next/link';
import { usePathname } from 'next/navigation';

export default function Navigation() {
  const pathname = usePathname();
  
  return (
    <nav>
      <Link 
        href="/" 
        className={pathname === '/' ? 'active' : ''}
      >
        Home
      </Link>
      
      <Link 
        href="/about" 
        className={pathname === '/about' ? 'active' : ''}
      >
        About
      </Link>
    </nav>
  );
}

4. Programmatic Navigation

'use client';
import { useRouter } from 'next/navigation';

export default function LoginPage() {
  const router = useRouter();
  
  const handleLogin = async () => {
    // Perform login logic
    const success = await loginUser();
    
    if (success) {
      // Navigate after successful login
      router.push('/dashboard');
    }
  };
  
  return (
    <button onClick={handleLogin}>
      Login
    </button>
  );
}

5. Navigation with Query Parameters

// Simple query string
<Link href="/search?q=nextjs&sort=recent">
  Search Next.js
</Link>

// Using object syntax
<Link href={{
  pathname: '/products',
  query: { 
    category: 'electronics', 
    sort: 'price',
    order: 'asc'
  }
}}>
  Electronics
</Link>
// Results in: /products?category=electronics&sort=price&order=asc
