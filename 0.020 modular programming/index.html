1. Export and Import a Function

// mathUtil.js
export function add(a, b) {
  return a + b;
}

// main.js
import { add } from './mathUtil.js';
console.log(add(5, 3));
Output: 8

2. Export Multiple Values

// info.js
export const name = 'Pramish';
export const age = 13;

// main.js
import { name, age } from './info.js';
console.log(`${name} is ${age} years old.`);
Output: Pramish is 13 years old.

3. Import All as an Object

// user.js
export const user = 'Admin';
export function greet() {
  return 'Hello User!';
}

// main.js
import * as data from './user.js';
console.log(data.user);
console.log(data.greet());
Output:
Admin
Hello User!

4. Default Export and Import

// greet.js
export default function greet(name) {
  return `Welcome, ${name}!`;
}

// main.js
import greet from './greet.js';
console.log(greet('Pramish'));
Output: Welcome, Pramish!

5. Rename Import

// calc.js
export function multiply(a, b) {
  return a * b;
}

// main.js
import { multiply as mul } from './calc.js';
console.log(mul(3, 4));
Output: 12

6. Export Object and Use

// config.js

  theme: 'dark',
  version: '1.0.0'
};

// main.js

console.log(settings.theme);
Output: dark

7. Export and Import JSON Data

Output: ["Aarav", "Pramish", "Sita"]

8. Combine Exports and Default

// util.js
export const PI = 3.14;
export default function area(r) {
  return PI * r * r;
}

// main.js
import area, { PI } from './util.js';
console.log(`Area: ${area(3)}, PI: ${PI}`);
Output: Area: 28.26, PI: 3.14
