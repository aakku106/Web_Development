3. Functions that Return Values

Calculate GST for any product price

Code:

function calculateGST(price, gstRate) {
    const gstAmount = (price * gstRate) / 100;
    const totalPrice = price + gstAmount;
    return {
        originalPrice: price,
        gstAmount: gstAmount,
        totalPrice: totalPrice
    };
}

6. Multiple Functions Working Together

Convert temperatures between Celsius and Fahrenheit

Code:

function celsiusToFahrenheit(celsius) {
    return (celsius * 9/5) + 32;
}

function fahrenheitToCelsius(fahrenheit) {
    return (fahrenheit - 32) * 5/9;
}

7. Real-World Function: EMI Calculator

Calculate monthly EMI for loans (Equated Monthly Installments)

Code:

function calculateEMI(principal, rate, years) {
    const monthlyRate = rate / (12 * 100);
    const months = years * 12;
    const emi = (principal * monthlyRate * Math.pow(1 + monthlyRate, months)) / 
                (Math.pow(1 + monthlyRate, months) - 1);
    return Math.round(emi);
}
